package it.comct.pmpay.services;

import java.io.BufferedWriter;
import java.io.File;
import java.io.FileWriter;
import java.io.StringReader;
import java.time.LocalDateTime;
import java.time.ZoneOffset;
import java.util.List;

import javax.xml.bind.JAXBContext;
import javax.xml.bind.JAXBException;
import javax.xml.bind.Unmarshaller;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Service;

import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.databind.ObjectWriter;

import it.comct.pmpay.models.AllineamentoPendenzeFullEsito;
import it.comct.pmpay.models.CSVHeader;
import it.comct.pmpay.models.EsitoAllinementoPendenza;

import it.comct.pmpay.models.LoginResponse;

import it.comct.pmpay.models.Pendenza;
import it.comct.pmpay.models.RicevutaResponse;
import it.comct.pmpay.models.UrlResponse;
import it.comct.pmpay.utils.HttpClient;
import it.comct.pmpay.utils.HttpGetRequest;
import it.comct.pmpay.utils.HttpPostRequest;

@Service
public class PMPayService {
	@Value("${pmpay.urladdress}")
	private String urlAddress;
	@Value("${pmpay.auth.token}")
	private String token;
	@Value("${pmpay.auth.user}")
	private String authUser;
	@Value("${pmpay.auth.password}")
	private String authPassword;
	@Value("${pmpay.urlOK}")
	private String urlOK;
	@Value("${pmpay.urlKO}")
	private String urlKO;
	@Value("${pmpay.urlS2S}")
	private String urlS2S;
	@Value("${pmpay.urlCancel}")
	private String urlCancel;

	Logger logger = LoggerFactory.getLogger(getClass().getName());

	
	public LoginResponse Autenticazione() throws Exception {
		LoginRequest l = new LoginRequest();
		l.setUsername(authUser);
		l.setPassword(authPassword);
		return Autenticazione(l);
	}

	public LoginResponse Autenticazione(LoginRequest login) throws Exception {
		HttpClient hc = new HttpClient(urlAddress + "/pagoparest/autenticazione");
		hc.PostExecute(login.toJson());

		int responseCode = hc.getResponseCode();
		if (responseCode == 200) {
			String response = hc.getResponse();
			ObjectMapper mapper = new ObjectMapper();
			LoginResponse auth = mapper.readValue(response, LoginResponse.class);
			auth.setStatus(200);
			return auth;
		}
		return null;
	}

	public String SavePagamento(String token, PagamentoAtteso pagamento) throws Exception {
		ObjectWriter ow = new ObjectMapper().writer().withDefaultPrettyPrinter();
		String json = ow.writeValueAsString(pagamento);
		logger.debug(json);
		HttpClient hc = new HttpClient(String.format("%s/pagoparest/ente/%s/pagamento", urlAddress,	pagamento.getAnagrafica().getCodiceEnte()));
		hc.setAuthorization("Bearer " + token);		
		hc.PostExecute(json);
		int responseCode = hc.getResponseCode();
		String response = hc.getResponse();
		if (responseCode != 201) { // Not Created
			logger.error(String.format("SavePagamento - codice errore %d \n %s", responseCode, response));
			return null;
		}
		ObjectMapper mapper = new ObjectMapper();
		IUVResponse r = mapper.readValue(response, IUVResponse.class);
		return r.getCode();
	}

	public byte[] OttieniBollettinoPagamento(String token, String ente, String iuv) throws Exception {
		HttpClient hc = new HttpClient(String.format("%s/pagoparest/ente/%s/pagamento/%s/bollettino", urlAddress, ente, iuv));
		hc.setAuthorization("Bearer " + token);
		byte [] ret = hc.GetExecuteAsByteArray();
		int responseCode = hc.getResponseCode();
		if (responseCode == 200) {
			return ret;
		}
		return null;
	}

	public UrlResponse GetUrlPagamento(String token, String ente, String iuv, Long idRichiesta) throws Exception {
		String b = String.format("{\"urlOk\":\"%s\",\"urlKo\":\"%s\",\"urlCancel\":\"%s\",\"idRichiesta\":\"%d\"}",
				urlOK, urlKO, urlCancel, LocalDateTime.now().toEpochSecond(ZoneOffset.UTC));
		logger.info(b);
		HttpClient hc = new HttpClient(String.format("%s/pagoparest/ente/%s/pagamento/%s/urlpagamento", urlAddress, ente, iuv));
		hc.setAuthorization("Bearer " + token);		
		hc.PostExecute(b);
		int responseCode = hc.getResponseCode();
		if (responseCode == 200) {
			String response = hc.getResponse();
			logger.debug(response);
			ObjectMapper mapper = new ObjectMapper();
			UrlResponse url = mapper.readValue(response, UrlResponse.class);
			url.setStatus(200);
			logger.info(url.getLocation());			
			return url;
		}
		return null;
	}

	public RicevutaResponse GetReceiptPayment(String token, String ente, String iuv) throws Exception {
		HttpClient hc = new HttpClient(String.format("%s/pagoparest/ente/%s/receipt/%s", urlAddress, ente, iuv));
		hc.setAuthorization("Bearer " + token);
		HttpGetRequest request = new HttpGetRequest();
		request.addParameter("idDebito", "false");
		hc.GetExecute(request);
		int responseCode = hc.getResponseCode();
		String response = hc.getResponse();
		RicevutaResponse ricevuta = null;
		if (responseCode == 200) {
			ObjectMapper mapper = new ObjectMapper();
			ricevuta = mapper.readValue(response, RicevutaResponse.class);
			ricevuta.setStatus(200);
		}else {
			ricevuta = new RicevutaResponse();
			ricevuta.setStatus(responseCode);
			ricevuta.setMessage(response);
		}
		return ricevuta;
	}
	
	public PagamentoResponse GetPaymentDetail(String token, String ente, String iuv) throws Exception {
		HttpClient hc = new HttpClient(String.format("%s/pagoparest/ente/%s/pagamento/%s", urlAddress, ente, iuv));
		hc.setAuthorization("Bearer " + token);
		HttpGetRequest request = new HttpGetRequest();
		request.addParameter("idDebito", "false");
		hc.GetExecute(request);
		int responseCode = hc.getResponseCode();
		if (responseCode == 200) {
			String response = hc.getResponse();
			ObjectMapper mapper = new ObjectMapper();
			PagamentoResponse pagamento = mapper.readValue(response, PagamentoResponse.class);
			pagamento.setStatus(200);
			return pagamento;
		}
		return null;
	}

}
